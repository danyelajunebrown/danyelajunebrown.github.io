        // Floating title class

        class FloatingTitle {

            constructor(project, index, container) {

                this.project = project;

                this.index = index;

                this.container = container;

                this.element = document.createElement('div');

                this.element.className = 'floating-title';

                this.element.textContent = project.title;

                

                this.x = Math.random() * (window.innerWidth - 300);

                this.y = Math.random() * (window.innerHeight - 200) + 100;

                

                this.vx = (Math.random() - 0.5) * 2;

                this.vy = (Math.random() - 0.5) * 2;

                

                this.element.style.left = this.x + 'px';

                this.element.style.top = this.y + 'px';

                

                // Hover effect for dummy works

                if (project.isDummy) {

                    this.element.addEventListener('mouseenter', () => {

                        this.element.textContent = project.hoverTitle;

                    });

                    this.element.addEventListener('mouseleave', () => {

                        this.element.textContent = project.title;

                    });

                }

                

                this.element.addEventListener('click', () => {

                    selectProject(this.index);

                });

                

                this.container.appendChild(this.element);

            }

            

            update() {

                if (activeSortAxis) {

                    this.updateConstrained();

                } else {

                    this.updateFree();

                }

                

                this.element.style.left = this.x + 'px';

                this.element.style.top = this.y + 'px';

            }

            

            updateFree() {

                this.x += this.vx;

                this.y += this.vy;

                

                if (this.x <= 0 || this.x >= window.innerWidth - this.element.offsetWidth) {

                    this.vx *= -1;

                    this.x = Math.max(0, Math.min(this.x, window.innerWidth - this.element.offsetWidth));

                }

                

                if (this.y <= 80 || this.y >= window.innerHeight - this.element.offsetHeight - 120) {

                    this.vy *= -1;

                    this.y = Math.max(80, Math.min(this.y, window.innerHeight - this.element.offsetHeight - 120));

                }

            }

            

            updateConstrained() {

                const score = this.getScore(activeSortAxis);

                let normalizedScore = score / 100;

                

                // Reverse if descending

                if (activeSortDirection === 'desc') {

                    normalizedScore = 1 - normalizedScore;

                }

                

                const targetX = (window.innerWidth - 300) * normalizedScore;

                

                this.x += (targetX - this.x) * 0.1;

                this.y += this.vy * 0.5;

                

                if (this.y <= 80 || this.y >= window.innerHeight - this.element.offsetHeight - 120) {

                    this.vy *= -1;

                    this.y = Math.max(80, Math.min(this.y, window.innerHeight - this.element.offsetHeight - 120));

                }

            }

            

            getScore(axis) {

                if (axis === 'chronology') {

                    const minYear = Math.min(...projects.filter(p => !p.isDummy).map(p => p.year));

                    const maxYear = Math.max(...projects.filter(p => !p.isDummy).map(p => p.year));

                    return ((this.project.year - minYear) / (maxYear - minYear)) * 100;

                }

                return this.project.scores?.[axis] || 50;

            }

        }

        async function uploadToGitHub(file) {

            const creds = getCredentials();

            const fileName = `assets/${Date.now()}-${file.name}`;

            const reader = new FileReader();

            

            return new Promise((resolve, reject) => {

                reader.onload = async () => {

                    const base64 = reader.result.split(',')[1];

                    

                    try {

                        const response = await fetch(`https://api.github.com/repos/${CONFIG.repo}/contents/${fileName}`, {

                            method: 'PUT',

                            headers: {

                                'Authorization': `token ${creds.githubToken}`,

                                'Content-Type': 'application/json'

                            },

                            body: JSON.stringify({

                                message: `Upload ${fileName}`,

                                content: base64,

                                branch: CONFIG.branch

                            })

                        });

                        

                        if (!response.ok) throw new Error('GitHub upload failed');

                        

                        const data = await response.json();

                        resolve(data.content.download_url);

                    } catch (error) {

                        reject(error);

                    }

                };

                

                reader.readAsDataURL(file);

            });

        }

        }


        // Scoring functionality

        function initializeScoring() {

            unscoredWorks = projects.filter(p => 

                !p.isDummy && 

                (p.scores.dearness === null || 

                 p.scores.completeness === null || 

                 p.scores.cost === null ||

                 p.scores.themes.length === 0 ||

                 p.scores.genres.length === 0)

            );

            

            // Shuffle unscored works

            unscoredWorks.sort(() => Math.random() - 0.5);

            

            currentScoringIndex = 0;

            loadNextWorkToScore();

            

            // Initialize checkboxes

            initializeCheckboxes();

        }


        function initializeCheckboxes() {

            const themesContainer = document.getElementById('themesCheckboxes');

            const genresContainer = document.getElementById('genresCheckboxes');

            

            themesContainer.innerHTML = THEMES.map(theme => `

                <label>

                    <input type="checkbox" value="${theme}">

                    ${theme}

                </label>

            `).join('');

            

            genresContainer.innerHTML = GENRES.map(genre => `

                <label>

                    <input type="checkbox" value="${genre}">

                    ${genre}

                </label>

            `).join('');

        }


        function loadNextWorkToScore() {

            if (currentScoringIndex >= unscoredWorks.length) {

                document.getElementById('currentWork').textContent = 'All works scored!';

                return;

            }

            

            const work = unscoredWorks[currentScoringIndex];

            document.getElementById('currentWork').textContent = `Score: "${work.title}"`;

            

            // Load current scores into checkboxes

            const themesCheckboxes = document.querySelectorAll('#themesCheckboxes input');

            themesCheckboxes.forEach(cb => {

                cb.checked = work.scores.themes.includes(cb.value);

            });

            

            const genresCheckboxes = document.querySelectorAll('#genresCheckboxes input');

            genresCheckboxes.forEach(cb => {

                cb.checked = work.scores.genres.includes(cb.value);

            });

            

            updateRelationalSlider();

        }


        function updateRelationalSlider() {

            const axis = document.getElementById('axisSelect').value;

            const work = unscoredWorks[currentScoringIndex];

            

            // Get already scored works on this axis

            const scoredWorks = projects

                .filter(p => !p.isDummy && p.scores[axis] !== null && p.id !== work.id)

                .sort((a, b) => a.scores[axis] - b.scores[axis]);

            

            const referenceContainer = document.getElementById('referenceWorks');

            const sliderTrack = document.getElementById('sliderTrack');

            

            // Clear previous markers

            sliderTrack.querySelectorAll('.slider-marker').forEach(m => m.remove());

            sliderTrack.querySelectorAll('.slider-marker-label').forEach(m => m.remove());

            

            if (scoredWorks.length === 0) {

                referenceContainer.innerHTML = '<div>No other works scored yet - place anywhere</div>';

            } else {

                referenceContainer.innerHTML = '<div>Already scored:</div>' + 

                    scoredWorks.map(w => `<div>• ${w.title} [${w.scores[axis]}%]</div>`).join('');

                

                // Add markers for scored works

                scoredWorks.forEach(w => {

                    const marker = document.createElement('div');

                    marker.className = 'slider-marker';

                    marker.style.left = w.scores[axis] + '%';

                    

                    const label = document.createElement('div');

                    label.className = 'slider-marker-label';

                    label.textContent = w.title.substring(0, 15) + '...';

                    label.style.left = w.scores[axis] + '%';

                    

                    sliderTrack.appendChild(marker);

                    sliderTrack.appendChild(label);

                });

            }

            

            // Set slider position

            const currentScore = work.scores[axis] || 50;

            document.getElementById('sliderHandle').style.left = currentScore + '%';

        }


        // Slider drag functionality

        let isDragging = false;

        const sliderHandle = document.getElementById('sliderHandle');

        const sliderTrack = document.getElementById('sliderTrack');


        sliderHandle.addEventListener('mousedown', () => isDragging = true);

        document.addEventListener('mouseup', () => isDragging = false);

        document.addEventListener('mousemove', (e) => {

            if (!isDragging) return;

            

            const rect = sliderTrack.getBoundingClientRect();

            let x = e.clientX - rect.left;

            x = Math.max(0, Math.min(x, rect.width));

            

            const percent = (x / rect.width) * 100;

            sliderHandle.style.left = percent + '%';

        });


        // Axis select change

        document.getElementById('axisSelect').addEventListener('change', updateRelationalSlider);


        // Save score button

        document.getElementById('saveScoreBtn').addEventListener('click', async () => {

            const work = unscoredWorks[currentScoringIndex];

            const axis = document.getElementById('axisSelect').value;

            

            // Get slider position

            const handleLeft = parseFloat(sliderHandle.style.left);

            work.scores[axis] = Math.round(handleLeft);

            

            // Get themes

            const themesChecked = Array.from(document.querySelectorAll('#themesCheckboxes input:checked'))

                .map(cb => cb.value);

            work.scores.themes = themesChecked;

            

            // Get genres

            const genresChecked = Array.from(document.querySelectorAll('#genresCheckboxes input:checked'))

                .map(cb => cb.value);

            work.scores.genres = genresChecked;

            

            await saveProjectsToGitHub();

            

            currentScoringIndex++;

            loadNextWorkToScore();

        });


        // Skip button

        document.getElementById('skipBtn').addEventListener('click', () => {

            currentScoringIndex++;

            loadNextWorkToScore();

        });


        // Bottom menu toggles - three states: off, ascending, descending

        let toggleStates = {

            chronology: null, // null, 'asc', 'desc'

            dearness: null,

            completeness: null,

            cost: null

        };


        document.querySelectorAll('.toggle-btn[data-axis]').forEach(btn => {

            btn.addEventListener('click', () => {

                const axis = btn.dataset.axis;

                const arrow = btn.querySelector('.arrow');

                

                // Cycle through states: null → asc → desc → null

                if (toggleStates[axis] === null) {

                    toggleStates[axis] = 'asc';

                    btn.classList.add('active');

                    arrow.textContent = '↑';

                    activeSortAxis = axis;

                    activeSortDirection = 'asc';

                } else if (toggleStates[axis] === 'asc') {

                    toggleStates[axis] = 'desc';

                    arrow.textContent = '↓';

                    activeSortDirection = 'desc';

                } else {

                    toggleStates[axis] = null;

                    btn.classList.remove('active');

                    activeSortAxis = null;

                    activeSortDirection = null;

                }

                

                // Deactivate other toggles

                if (toggleStates[axis] !== null) {

                    document.querySelectorAll('.toggle-btn[data-axis]').forEach(b => {

                        if (b !== btn) {

                            const otherAxis = b.dataset.axis;

                            toggleStates[otherAxis] = null;

                            b.classList.remove('active');

                        }

                    });

                }

            });

        });

                    label.innerHTML = `<input type="checkbox" value="${genre}"> ${genre}`;

                    label.querySelector('input').addEventListener('change', applyFilters);

                    dropdown.appendChild(label);

                });

            }

        });

        });


        function applyFilters() {

            const selectedThemes = Array.from(document.querySelectorAll('#themesDropdown input:checked'))

                .map(cb => cb.value);

            const selectedGenres = Array.from(document.querySelectorAll('#genresDropdown input:checked'))

                .map(cb => cb.value);

            

            if (selectedThemes.length === 0 && selectedGenres.length === 0) {

                // No filters, show all

                floatingElements.forEach(el => {

                    el.element.classList.remove('hidden');

                });

                return;

            }

            

            // Apply filters and snap to grid

            floatingElements.forEach((el, index) => {

                const project = projects[index];

                

                const matchesTheme = selectedThemes.length === 0 || 

                    selectedThemes.some(t => project.scores.themes.includes(t));

                const matchesGenre = selectedGenres.length === 0 || 

                    selectedGenres.some(g => project.scores.genres.includes(g));

                

                if (matchesTheme && matchesGenre) {

                    el.element.classList.remove('hidden');

                } else {

                    el.element.classList.add('hidden');

                }

            });

            

            // Snap visible elements to grid

            snapToGrid();

        }


        function snapToGrid() {

            const visible = floatingElements.filter(el => !el.element.classList.contains('hidden'));

            const cols = Math.ceil(Math.sqrt(visible.length));

            const cellWidth = window.innerWidth / cols;

            const cellHeight = (window.innerHeight - 200) / Math.ceil(visible.length / cols);

            

            visible.forEach((el, i) => {

                const row = Math.floor(i / cols);

                const col = i % cols;

                

                el.x = col * cellWidth + cellWidth / 2 - 150;

                el.y = row * cellHeight + cellHeight / 2 + 100;

                el.vx = 0;

                el.vy = 0;

            });

        }

            document.querySelectorAll('#themesDropdown input, #genresDropdown input').forEach(cb => cb.checked = false);

            

            floatingElements.forEach(el => {

                el.element.classList.remove('hidden');

                el.vx = (Math.random() - 0.5) * 2;

                el.vy = (Math.random() - 0.5) * 2;

            });

        });
